安装ActiveMQ ​

yaml
version: '3'
services:
  activemq:
    image: webcenter/activemq                    # 镜像`webcenter/activemq`
    container_name: activemq                     # 容器名为'activemq'
    hostname: activemq                         # 定义主机名
    restart: always
    #容器的映射端口
    ports:
      - 61613:61613
      - 61616:61616
      - 8161:8161
    volumes:   # 数据卷挂载路径设置,将本机目录映射到容器目录
      # 宿主机时间
      - /etc/localtime:/etc/localtime
      - /application/dockerData/activemq/data/activemq:/data/activemq
      - /application/dockerData/activemq/var/log/activemq:/var/log/activemq
    environment:                                # 设置环境变量,相当于docker run命令中的-e
      TZ: Asia/Shanghai
      LANG: en_US.UTF-8
      ACTIVEMQ_ADMIN_LOGIN: admin      # 登录账号
      ACTIVEMQ_ADMIN_PASSWORD: admin   # 登录密码
      ACTIVEMQ_CONFIG_MINMEMORY: 512
      ACTIVEMQ_CONFIG_MAXMEMORY: 2048
